# HTOP - Диспетчер завдань, процесів

# КОМАНДИ

  htop - запуск
  echo - $! - перегляд останнього PID процесу\
  F5/t - відображення дерева команд процесу
  M - сортування по памяті
  P - сортування по процесору
  /+команда - пошук процесу
  \+команда - фільтр процесу
  u - фільт процесів по user
  k+9 - завершення процесу
  процес+e - змінне оточення процесу
  пробіл - виділення процесів
  U - скидання виділення процесу
  F2 - налаштування

# ІНТЕРФЕЙС

  cpu - кількість ядер та навантаження на процесор
  mem - загальна кількість та навантаження на оперативну память
  tasks - кількість запущених процесів
  runing - запущені процеси
  thr(threads) - потоки
  kthr(kernel threads) - потоки ядра
  load average - середня загрузка системи протягом за останні 1, 5, 15 хвилин
  uptime - час безперебійної роботи системи

  pid - id процеса
  user - користувач котрий запустив процес
  pri - пріоритет процесу (чим нижче значення тим вище пріоритет)
  cpu - кількість споживання в відсотках процесора
  mem - кількість споживання в відсотках пам'яті
  command - список команд котрі запущенні в системі

# УТИЛІТИ

  lsof означає " LiSt Open Files ", використовується для того,
  щоб зрозуміти, які файли відкриті яким процесом.
  
  # команди утиліти

  l - перегляд файлів відкритим процесом
